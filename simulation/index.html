How It Works

Closed-loop
light tracking

Four LDR sensors compare light intensity at each corner of the panel. The Arduino reads all four simultaneously, calculates the differential, and nudges the servos in the direction of greater brightness — converging on perfect alignment with the sun.

STEP 01

Sense

Four LDR sensors at the panel corners read ambient light intensity as analog voltages (0–1023) via voltage dividers.

STEP 02

Compare

Arduino averages top vs bottom for tilt error, and left vs right for pan error. A 15-unit deadband prevents jitter.

STEP 03

Correct

Servo positions increment 1° per cycle toward the brighter side — smooth, stable tracking that locks onto the sun.

Arduino
Source Code

Full production code — upload directly to your UNO.

solar_tracker.ino 
// ============================================================
//  Dual-Axis Solar Tracker
//  Board  : Arduino UNO R3
//  Built  : Mangaluru, Karnataka, India
// ============================================================

#include <Servo.h>

// Servo objects
Servo panServo;
Servo tiltServo;

// Pin definitions
const int LDR_NW = A0;
const int LDR_NE = A1;
const int LDR_SW = A2;
const int LDR_SE = A3;
const int PAN_PIN  = 9;
const int TILT_PIN = 10;

// Config
const int SERVO_MIN  = 10;
const int SERVO_MAX  = 170;
const int THRESHOLD  = 15;
const int SERVO_STEP = 1;
const int LOOP_DELAY = 20;

int panPos  = 90;
int tiltPos = 90;

void setup() {
  Serial.begin(9600);
  panServo.attach(PAN_PIN);
  tiltServo.attach(TILT_PIN);
  panServo.write(panPos);
  tiltServo.write(tiltPos);
  Serial.println("Solar Tracker Ready");
}

void loop() {
  int nw = analogRead(LDR_NW);
  int ne = analogRead(LDR_NE);
  int sw = analogRead(LDR_SW);
  int se = analogRead(LDR_SE);

  int avgTop    = (nw + ne) / 2;
  int avgBottom = (sw + se) / 2;
  int avgLeft   = (nw + sw) / 2;
  int avgRight  = (ne + se) / 2;

  int tiltDiff = avgTop - avgBottom;
  int panDiff  = avgLeft - avgRight;

  // Tilt correction
  if (abs(tiltDiff) > THRESHOLD) {
    tiltPos = constrain(
      tiltPos + (tiltDiff > 0 ? SERVO_STEP : -SERVO_STEP),
      SERVO_MIN, SERVO_MAX
    );
    tiltServo.write(tiltPos);
  }

  // Pan correction
  if (abs(panDiff) > THRESHOLD) {
    panPos = constrain(
      panPos + (panDiff > 0 ? -SERVO_STEP : SERVO_STEP),
      SERVO_MIN, SERVO_MAX
    );
    panServo.write(panPos);
  }

  Serial.print("NW:"); Serial.print(nw);
  Serial.print(" NE:"); Serial.print(ne);
  Serial.print(" PAN:"); Serial.print(panPos);
  Serial.print("° TILT:"); Serial.print(tiltPos);
  Serial.println("°");

  delay(LOOP_DELAY);
}
